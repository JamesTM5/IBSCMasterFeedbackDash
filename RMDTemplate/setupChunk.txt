files = list.files(path = paste(".", "data", sep = "/"), pattern = '^.*rds$')
fileList <- list()
for (i in 1:length(files)) {
  file <- readRDS(paste("./data", files[i], sep = "/"))
  name <- as.character(files[i])
  fileList[[name]] <- file
}

#set an arbitary seed
set.seed (fileList[[1]]$seed)
